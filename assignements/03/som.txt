# STAGE 1: Builder
# Use standard Python image to ensure path consistency with runtime
FROM python:3.12-slim-bookworm AS builder

# Install uv by copying the binary from the official image
COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /bin/

WORKDIR /app

# Enable bytecode compilation
ENV UV_COMPILE_BYTECODE=1

# Copy dependency files
COPY pyproject.toml uv.lock ./

# Install dependencies
# --frozen: Use exact versions from lockfile
# --no-dev: Exclude development dependencies (pytest, etc.)
# --no-install-project: Do not install the project itself yet, just dependencies
RUN uv sync --frozen --no-dev --no-install-project

# STAGE 2: Runtime
# Must match the builder image exactly
FROM python:3.12-slim-bookworm

WORKDIR /app

# Copy the virtual environment from the builder
COPY --from=builder /app/.venv /app/.venv

# Copy the application source code
COPY src ./src

# Set environment variables
# Add the virtual env to PATH so we can run installed binaries
ENV PATH="/app/.venv/bin:$PATH"
ENV PYTHONPATH="/app/src"

# Create a non-root user for security
RUN useradd -m fireguard && chown -R fireguard /app
USER fireguard

# Expose the port
EXPOSE 8000

# Use 'python -m uvicorn' to ensure we use the venv's python interpreter
CMD ["python", "-m", "uvicorn", "src.app.main:app", "--host", "0.0.0.0", "--port", "8000"]
