# STAGE 1: Builder
FROM python:3.12-slim-bookworm AS builder
COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /bin/

WORKDIR /app
ENV UV_COMPILE_BYTECODE=1

# Copy dependency files
COPY pyproject.toml uv.lock ./

# Install dependencies
# We assume uvicorn is in pyproject.toml
RUN uv sync --frozen --no-dev --no-install-project

# STAGE 2: Runtime
FROM python:3.12-slim-bookworm

WORKDIR /app

# Copy the virtual environment
COPY --from=builder /app/.venv /app/.venv

# Copy application code
COPY src ./src

# Create user
RUN useradd -m fireguard && chown -R fireguard /app
USER fireguard

# Expose port
EXPOSE 8000

# FIX: Use the ABSOLUTE path to the uvicorn binary inside the .venv
# This guarantees we use the correct environment.
CMD ["/app/.venv/bin/uvicorn", "src.app.main:app", "--host", "0.0.0.0", "--port", "8000"]
